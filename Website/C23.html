C program to use pre-processor commands.

#include< conio.h >

}


#include< stdio.h >
#include< conio.h >
#define pi 3.14
#undef pi
int main(){
int r;
printf("Enter radius : ");
scanf("%d",&r);
printf("Area of circle is : %f",3.14*3.14*r);

}
