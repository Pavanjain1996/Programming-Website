Java program to find LCM of two numbers.

import java.util.Scanner;
class Example{
public static void main(String []args){
int i,a,b,lar;
Scanner sc=new Scanner(System.in);
System.out.println("Enter two numbers");
a=sc.nextInt();
b=sc.nextInt();
if(a>b)
lar=a;
else
lar=b;
for(i=lar;i<=a*b;i++){
if(i%a==0 && i%b==0){
break;
}
}
System.out.println("The LCM is : "+i);
}
}
