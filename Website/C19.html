C program to swap two numbers using pointers.

#include< stdio.h >
#include< conio.h >
int main(){
int a,b,*p,*q;
printf("Enter two numbers : ");
scanf("%d%d",&a,&b);
p=&a;
q=&b;
printf("Before swapping\na = %d b = %d",*p,*q);
*p=*p+*q;
*q=*p-*q;
*p=*p-*q;
printf("\nAfter swapping\na = %d b = %d",*p,*q);
getch();
}
