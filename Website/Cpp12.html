Cpp program to find LCM of two numbers.

#include< iostream >
#include< conio.h >
using namespace std;
int big(int a,int b){
if(a>b)
return(a);
else
return(b);
}
int main(){
system("cls");
int n1,n2,i,lcm;
cout<<"Enter two numbers ";
cin>>n1>>n2;
for(i=big(n1,n2);i<=(n1*n2);i++){
if(i%n1==0 && i%n2==0){
lcm=i;
break;
}
}
cout<<"L.C.M. of "<< n1<<" and "<< n2<<" is "<< i;
getch();
}
