Cpp program to calculate nCr and nPr.

#include< iostream >
#include< conio.h >
using namespace std;
long int fact(int n){
if(n==0 || n==1)
return(1);
else
return(n*fact(n-1));
}
main(){
int ch,n,r;
cout<<"\n\t1.nPr\n\t2.nCr\n\t3.Exit\n\tEnter your choice : ";
cin>>ch;
if(n>=r){
switch(ch){
case 1:
cout<<"\n\tEnter the value of n and r : ";
cin>>n>>r;
cout<<"\n\t"<< n<<" P "<< r<<" = "<< fact(n)/fact(n-r);
break;
case 2:
cout<<"\n\tEnter the value of n and r : ";
cin>>n>>r;
cout<<"\n\t"<< n<<" C "<< r<<" = "<< fact(n)/(fact(n-r)*fact(r));
break;
case 3:
exit(0);
}
}
getch();
}
