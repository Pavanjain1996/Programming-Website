Java program to find HCF of two numbers.

import java.util.Scanner;
class Example{
public static void main(String []args){
int i,a,b,small;
Scanner sc=new Scanner(System.in);
System.out.println("Enter two numbers"0);
a=sc.nextInt();
b=sc.nextInt();
if(a>b)
small=b;
else
small=a;
for(i=small;i>0;i--){
if(a%i==0 && b%i==0){
break;
}
}
System.out.println("The HCF is : "+i);
}
}
