Cpp program to remove adjacent duplicate characters from a string.

#include< iostream >
#include< conio.h >
using namespace std;
char* remove_duplicate(char *s){
int i,j;
for(i=0;s[i];i++){
if(s[i]==s[i+1]){
for(j=i+1;s[j];j++){
s[j]=s[j+1];
}
i=-1;
}
}
return(s);
}
main(){
char arr[20];
cout<<"Enter a string : ";
cin.getline(arr,19);
cout<<"\n\nAfter removing duplicate characters string is\n\n";
cout<< remove_duplicate(arr);

}
