C program to find the fibonacci series till n terms.

#include< stdio.h >
#include< conio.h >
int fibo(int n){
if(n<=2)
return 1;
else
return fibo(n-1)+fibo(n-2);
}
int main(){
int a;
printf("Which term of fibonacci series you want to find ? ");
scanf("%d",&a);
printf("%dth term of fibonacci series is : %d ",a,fibo(a));
getch();
}
