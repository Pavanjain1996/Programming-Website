Cpp program to find out cricket batting averages.

#include< iostream >
#include< iomanip >
#include< cstdlib >
using namespace std;
class batsman{
char name[10];
int runs;
int innings;
int tno;
float avg;
public:
void getData(){
cout<<"Enter name : ";
cin>>name;
fflush(stdin);
cout<<"Enter runs : ";
cin>>runs;
cout<<"Enter innings : ";
cin>>innings;
cout<<"Enter innings not out : ";
cin>>tno;
}
void calcAvg(){
avg=(float)runs/(float)(innings-tno);
}
void showData(){
cout<< setw(10)<< name;
cout<< setw(10)<< runs;
cout<< setw(10)<< innings;
cout<< setw(10)<< tno;
cout<< setw(10)<< avg;
cout<< endl;
}
};
main(){
batsman b1,b2,b3;
b1.getData();
b2.getData();
b3.getData();
b1.calcAvg();
b2.calcAvg();
b3.calcAvg();
cout<<"\nThe Details are\n\n";
b1.showData();
b2.showData();
b3.showData();
}
