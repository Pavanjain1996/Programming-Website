Python program to make a simple calculator.

def add(a,b):
print("Addition is "+str(a+b))

def subtract(a,b):
print("Subtraction is "+str(a-b))

def multiply(a,b):
print("Multiplacation is "+str(a*b))

def divide(a,b):
print("Division is "+str(a/b))

a=int(input("Enter first number : "))
b=int(input("Enter second number : "))
print("1.Addition")
print("2.Subtraction")
print("3.Multiplacation")
print("4.Division")
n=int(input("Enter your choice : "))
if n==1:
add(a,b)

elif n==2:
subtract(a,b)

elif n==3:
multiply(a,b)

elif n==4:
divide(a,b)

else:
print("Wrong Choice")
