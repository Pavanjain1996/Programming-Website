Cpp program to calculate sin(theta).

#include< iostream >
#include< cmath >
using namespace std;
long int fact(int num){
long int fac=1;
while(num>0){
fac=fac*num;
num--;
}
return fac;
}
main(){
cout<<"Enter the value of x : ";
float x,nume,term,sum=0;
int i,deg;
cin>>x;
for(i=0;i<5;i++){
deg=(2*i)+1;
nume=pow(x,deg);
term=nume/fact(deg);
if(i%2==0)
sum=sum+term;
else
sum=sum-term;
}
cout<<"sin "<< x<<" = "<< sum;
}
