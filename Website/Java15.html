Java program to convert temperature from farhenheit to celcius.

import java.util.Scanner;
class Example{
public static void main(String []args){
double tempc,tempf;
System.out.print("Enter temperature in farhenheit : ");
Scanner sc=new Scanner(System.in);
tempf=sc.nextDouble();
tempc=5*(tempf-32)/9.0;
System.out.print("Temperature in celcius is : "+tempc);
}
}
