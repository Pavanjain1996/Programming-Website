C program to use structure.

#include< stdio.h >
#include< conio.h >
struct Student{
char name[10];
int age;
char gender;
};
int main(){
struct Student s1;
printf("Enter Name : ");
gets(s1.name);
printf("Enter age : ");
scanf("%d",&s1.age);
fflush(stdin);
printf("Enter gender : ");
scanf("%c",&s1.gender);
printf("\nName : %s",s1.name);
printf("\nAge : %d",s1.age);
printf("\nGender : %c",s1.gender);
getch();
}
