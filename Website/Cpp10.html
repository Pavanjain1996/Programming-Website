Cpp program to find GCD or HCF.

#include< iostream >
#include< conio.h >
using namespace std;
int big(int a,int b){
if(a>b)
return(a);
else
return(b);
}
int main(){
system("cls");
int n1,n2,i,hcf;
cout<<"Enter two numbers ";
cin>>n1>>n2;
for(i=1;i<=big(n1,n2);i++){
if(n1%i==0 && n2%i==0){
hcf=i;
}
}
cout<<"H.C.F. of "<< n1<<" and "<< n2<<" is "<< hcf;
getch();
}
