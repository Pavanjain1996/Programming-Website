Java program to multiply two matrices.

import java.util.Scanner;
class Example{
public static void main(String []args){
int a[][]=new int[3][3];
int b[][]=new int[3][3];
int c[][]=new int[3][3];
int i,j;
Scanner sc=new Scanner(System.in);
System.out.println("Enter Matrix 1 : ");
for(i=0;i<3;i++){
for(j=0;j<3;j++){
a[i][j]=sc.nextInt();
}
}
System.out.println("Enter Matrix 2 : ");
for(i=0;i<3;i++){
for(j=0;j<3;j++){
b[i][j]=sc.nextInt();
}
}
for(i=0;i<3;i++){
for(j=0;j<3;j++){
c[i][j]=a[i][0]*b[0][j]+a[i][1]*b[1][j]+a[i][2]*b[2][j];
}
}
System.out.println("The multiplacation matrix is");
for(i=0;i<3;i++){
for(j=0;j<3;j++){
System.out.print(""+c[i][j]+" ");
}
System.out.println();
}
}
}
