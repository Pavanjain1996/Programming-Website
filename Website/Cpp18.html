Cpp program to generate random numbers.

#include< Windows.h >
#include< iostream >
#include< conio.h >
using namespace std;
int sysRan(int n){
SYSTEMTIME t;
GetSystemTime(&t);
int ranVal;
ranVal=(t.wDay+t.wDayOfWeek+t.wHour+t.wMilliseconds
+t.wMinute+t.wMonth+t.wSecond+t.wYear)%n;
return(ranVal);
}
main(){
system("cls");
while(!kbhit()){
int macNo=sysRan(10000);
cout<<"Random Number = "<< macNo<< endl;
Sleep(1000);
}
getch();
}
