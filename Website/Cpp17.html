Cpp program to check whether a number is prime or not.

#include< iostream >
#include< conio.h >
using namespace std;
int main(){
system("cls");
int n,i,flag=0;
cout<<"Enter a Number : ";
cin>>n;
if(n==1)
cout<<"1 is not prime";
else{
for(i=2;i<=n/2;i++){
if(n%i==0){
flag=1;
break;
}
}
if(flag==0)
cout<< n<<" is prime";
else
cout<< n<<" is not prime";
}
getch();
}
