Cpp program to convert temperature fron farhenheit to celcius.

#include< iostream >
using namespace std;
class temp{
float tempf;
public:
void setTemp(){
cin>>tempf;
}
float toCelcius(){
float tempc=(tempf-32)*5.0/9.0;
return tempc;
}
};
main(){
temp t1;
cout<<"Enter temperature in farhenhiet : ";
t1.setTemp();
cout<<"\nThe temperature is celcius is : "<< t1.toCelcius();
}
